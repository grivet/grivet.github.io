shell: errno memo

2021-04-07 
# $1: (nil) or errno value
# output: full errno list or errno value name
errno () {(
    err=${1}

    [ "$err" ] && filter="| grep \"${err}\""
    filter="$filter |column -t"
    cmdline="cc -E -dM /usr/include/errno.h \
            |grep 'define E' \
            |sort -n -k 3 \
            |cut -d' ' -f2- \
            $filter"
    eval $cmdline
)}

$ errno 26
ETXTBSY  26
ENOKEY   126
$ errno ENOBUFS
ENOBUFS  105
